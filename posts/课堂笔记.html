DTD文档类型定义（实体）

内部实体

定义一个内部实体（类似于变量名/变量值）

<!ENTITY 实体名称 “实体的值”>

<?xml version="1.0"?>
<!DOCTYPE note[
<!ELEMENT note (name)>
<!ENTITY hack3r "hu3sky">
]>
<note>
  <name>&hack3r;</name>
</note>

外部实体（调用外部DTD文件）

<!ENTITY 实体名称 SYSTEM “URI/URL”>

<?xml version="1.0"?>
<!DOCTYPE a[
<!ENTITY  name SYSTEM "http://xx.com/aa.dtd">]>
<a>
<name>&name;</name>
</a>

//aa.dtd文件
<!DOCTYPE scan [
<!ENTITY test SYSTEM "file:///c:/1.txt">]>  //DTD部分 从外部获取文件（实体file:///c:/1.txt并命名为test

//&实体名;  调用实体
<scan>&test;</scan>  //XML部分，实体test在<scan></scan>标签内有用

参数实体

<!ENTITY %实体名称 “值”> 

<?xml version="1.0"?>
<!DOCTYPE note [
<!ENTITY % b "<!ENTITY b1 'awsl'>">
        %b;
        ]>
<note>&b1;</note>

外部参数实体

<!ENTITY %实体名称 SYSTEM “URL”>

<?xml version=”1.0”?>
<!DOCTYPE a[
<!ENTITY % name SYSTEM  “file:///etc/passwd”> //外部实体调用伪协议
%name;
]>
